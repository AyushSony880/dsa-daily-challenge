36. Buble sort

```js
let arr = [5, 0, 1, 2, 7, 9, 3, 4];

function bubbleSort(arr) {
  let count = 0;
  for (let i = 0; i < arr.length - 1; i++) {
    let isRun = false;
    for (let j = 0; j < arr.length - i - 1; j++) {
      ++count;
      let temp;
      if (arr[j] > arr[j + 1]) {
        isRun = true;
        temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp;
      }
    }
    if (!isRun) {
      console.log(count);
      return arr;
    }
  }
  console.log(count);
  return arr;
}

console.log(bubbleSort(arr));
```

37. selection sort

```js
// let arr = [5, 3, 4, 8, 9, 2, 7];
let arr = [2, 3, 4, 5, 7, 8, 9];

function selectionSort(arr) {
  let count = 0;
  for (let i = 0; i < arr.length - 1; i++) {
    let smallestNum = i;
    for (let j = i + 1; j < arr.length; j++) {
      ++count;
      if (arr[j] < arr[smallestNum]) {
        smallestNum = j;
      }
    }

    if (smallestNum != i) {
      let temp = arr[i];
      arr[i] = arr[smallestNum];
      arr[smallestNum] = temp;
    }
  }
  console.log(count);

  return arr;
}

console.log(selectionSort(arr));
```
