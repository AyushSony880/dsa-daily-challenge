36. Buble sort

```js
let arr = [5, 0, 1, 2, 7, 9, 3, 4];

function bubbleSort(arr) {
  let count = 0;
  for (let i = 0; i < arr.length - 1; i++) {
    let isRun = false;
    for (let j = 0; j < arr.length - i - 1; j++) {
      ++count;
      let temp;
      if (arr[j] > arr[j + 1]) {
        isRun = true;
        temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp;
      }
    }
    if (!isRun) {
      console.log(count);
      return arr;
    }
  }
  console.log(count);
  return arr;
}

console.log(bubbleSort(arr));
```
