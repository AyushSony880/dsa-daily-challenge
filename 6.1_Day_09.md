34. linear search

```js
function linearSearch(arr, val) {
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] === val) return i;
  }
  return -1;
}

console.log(linearSearch([1, 2, 5, 6, 4, 8], 5));
```

35. Binary search

```js
function BinarySearch(nums, target) {
  let L = 0,
    R = Math.floor(nums.length - 1),
    M;

  while (L <= R) {
    M = Math.floor((L + R) / 2);
    if (nums[M] === target) {
      return M;
    } else if (nums[M] < target) {
      L = M + 1;
    } else {
      R = M - 1;
    }
  }
  return -1;
}
console.log(BinarySearch([1, 2, 5, 6, 4, 8], 5));
```
