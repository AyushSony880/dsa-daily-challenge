14. Write a function that returns the count of digits in a number.

```js
function numOfDigits(num) {
  let count = 0;
  if (num === 0) return 1;
  num = Math.abs(num);
  //        OR
  // if (num < 0)    num *= -1;
  while (num > 0) {
    num = Math.floor(num / 10);
    count++;
  }
  return count;
}

console.log(numOfDigits(0));
```

15. Write a function that true if number is palindrome(121,151,11 all are palindrome but 123,12 is not palindrome) and false if number is not palindrome.And return false for negative number.

```js
function findPalindrome(num) {
  if (num < 0) return false;
  let originalNum = num;
  let newNum = 0;
  while (num > 0) {
    newNum = newNum * 10 + (num % 10);
    num = num = Math.floor(num / 10); // OR (num - (num % 10)) / 10;
  }
  return originalNum === newNum;
  //                OR
  //   if (originalNum === newNum) return true;
  //   else return false;
}

console.log(findPalindrome(12));
```

16. Write a function that reverse the given integer,Only for 32-bit integers range[ -(2^{31}),(2^{31}-1)]

```js
function reverseInt(num) {
  let NumCopy = num;
  num = Math.abs(num);

  let revNum = 0;
  while (num > 0) {
    revNum = revNum * 10 + (num % 10);
    num = Math.floor(num / 10);
  }
  let limit = Math.pow(2, 31);
  if (revNum < -limit || revNum > limit) return 0;
  if (NumCopy < 0) revNum *= -1;
  return revNum;
}

console.log(reverseInt(1235));
```
