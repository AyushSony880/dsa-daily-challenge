1.Write a function that searches for an element in an array and return the index, if the element is not present then just return -1.

```js
let arr = [55, 12, 56, 66, 98, 45, 91, 57];
function findElementIndex(arr, elem) {
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] === elem) return i;
  }
  return -1;
}

console.log(findElementIndex(arr, 58));
```

2. Write a function that returns the number of negative number in an array.

```js
let arr = [-9, -54, 7, 97, -78, 56, 12, -74, 58];
function countNegativeNumber(arr) {
  let negativeCount = 0;
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] < 0) ++negativeCount;
  }
  return negativeCount;
}
console.log(countNegativeNumber(arr));
```

3. Write a function that returns the largest number in the array.

```js

function findLargestNum(arr) {
  let largestNum = arr[0];
  for (let i = 1; i < arr.length; i++) {
    if (arr[i] > largestNum) largestNum = arr[i];
  }
  return largestNum;
}
console.log(findLargestNum(arr));
```

```js
  if (arr.length < 2) {
```

11. print the pattern.

\_ \_ \_ _ 1
_ \_ _ 1 1
_ _ 1 1 1
_ 1 1 1 1
1 1 1 1 1

```js
for (let i = 0; i < 5; i++) {
  let row = "";
  for (let j = 0; j < 5; j++) {
    if (j < 5 - (i + 1)) {
      row += " ";
    } else row += "*";
  }
  console.log(row);
}
```
