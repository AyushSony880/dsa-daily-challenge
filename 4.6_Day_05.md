22. Given an integer array nums, move all 0's to the end of it while maintaining the relative order of the non-zero elements.

Note: that you must do this in-place without making a copy of the array.

```js
function moveZeroes(nums) {
  let x = 0;
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] !== 0) {
      nums[x] = nums[i] + nums[x];
      nums[i] = nums[x] - nums[i];
      nums[x] = nums[x] - nums[i];
      x++;
    }
  }
  return nums;
}
console.log(moveZeroes([0, 0, 0, 1, 2, 3, 0, 0, 4, 0, 0, 5, 0, 0, 6]));
```

23. Given a binary array nums, return the maximum number of consecutive 1's in the array.

```js
function findMaxConsecutiveOnes(nums) {
  let maxConsecutive = 0;
  let consecutive = 0;
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] !== 1) consecutive = 0;
    else consecutive++;
    if (consecutive > maxConsecutive) maxConsecutive = consecutive;
  }
  return maxConsecutive;
}
console.log(findMaxConsecutiveOnes([0, 1, 1, 0, 0, 0, 0, 1]));
```

**1st approach**

```js
function missingNumber(nums) {
  let totalSum = 0;
  let actualSum = 0;
  for (let i = 1; i <= nums.length; i++) {
    actualSum += i;
    totalSum += nums[i - 1];

}
```
