38. Insertion sort

```js
function insertionSort(a) {
  for (let i = 1; i < a.length; i++) {
    let s = i - 1,
      c = i,
      temp = a[i];
    for (let j = s; j >= 0; j--) {
      if (a[c] < a[j]) {
        a[c] = a[j];
        a[j] = temp;
        c--;
      } else break;
    }
  }

  return a;
}

console.log(insertionSort([3, 4, 5, 7, 8, -9, 2]));
```
