39. Merge sort

```js
function mergeTwoSortedArr(a1, a2) {
  let m = a1.length;
  let n = a2.length;
  let a3 = [];
  for (let i = m + n - 1; i >= 0; i--) {
    if ((a1[m - 1] < a2[n - 1] && n >= 0) || m <= 0) {
      a3[i] = a2[n - 1];
      n--;
    } else {
      a3[i] = a1[m - 1];
      m--;
    }
  }
  return a3;
}

function mergeSort(a) {
  if (a.length === 1) {
    return a;
  }
  let mid = Math.floor(a.length / 2);
  let left = mergeSort(a.slice(0, mid));
  let right = mergeSort(a.slice(mid));
  return mergeTwoSortedArr(left, right);
}

console.log(mergeSort([1, 5, 8, 7, 2, 4, 3, 9]));
```
