28. Write a recursive function to find sum of n number.

```js
function fun(n) {
  if (n < 1) return 0;

  return n + fun(--n);
}

console.log(fun(5));
```

29. Write a recursive function to find sum of array element.

```js
let arr = [1, 2, 3, 4, 5, 6];

function sum(n) {
  if (n === 0) return arr[n];
  return arr[n] + sum(n - 1);
}

console.log(sum(arr.length - 1));
```

30. Write a recursive function to find sum of all odd number in an array.

```js
let arr = [1, 2, 3, 4, 5, 6];

function sum(n) {
  let isOdd = arr[n] % 2 !== 0;
  if (n === 0) return isOdd ? arr[n] : 0;
  return (isOdd ? arr[n] : 0) + sum(n - 1);
}

console.log(sum(arr.length - 1));
```
